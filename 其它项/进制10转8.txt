#include<iostream>
using namespace std;
int main()
{
	int num10, temp,save=0,next=1;
	cout << "输入10进制数字：";cin >> num10;
	while (true){
		temp=num10 % 8;save += temp * next;num10 /= 8;next*=10;
		if (num10 <= 8){save += num10 * next;break;}
	}
	cout << "转换为8进制：" << save << endl;
	system("pause");return 0;
}
