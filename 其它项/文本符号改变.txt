#include<iostream>
using namespace std;
#include<fstream>
#include<string>
#include<conio.h>
#include <Windows.h>
int main()
{

	cout << "\033[31m";
	ifstream ifs;
	ifs.open("字符序列01.txt");
	if (!ifs.is_open())
	{
		cout<<"打开失败！" << endl;
	}
	string str;
	while (getline(ifs, str))
	{
		char s[1024];
		for (int i = 0; i < str.size(); i++)
		{
			if (str[i] == '@')
			{
				s[i] = ' ';
			}
			else
			{
				s[i] = str[i];
			}

			cout << s[i];
		}
		cout << endl;


	}




	system("pause");
	return 0;
}
