void Time()
{
	clock_t startTime, endTime;//计时
	double Time = 0;
	while (Time < 0.04)
	{
		startTime = clock();//计时开始

		endTime = clock();//计时结束
		Time += (double)(endTime - startTime) / CLOCKS_PER_SEC;
	}
}
string to_String(int n)
{
	int m = n;char s[max];char ss[max];
	int i = 0, j = 0;
	if (n < 0){
		m = 0 - m;j = 1;ss[0] = '-';
	}
	while (m > 0){
		s[i++] = m % 10 + '0';m /= 10;
	}
	s[i] = '\0';i = i - 1;
	while (i >= 0){
		ss[j++] = s[i--];
	}
	ss[j] = '\0';
	return ss;
}
